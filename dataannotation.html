function parseGoogleDoc() { // Open the Google Document by its ID var docId = "YOUR_DOCUMENT_ID"; // Replace with your Google Doc ID var doc = DocumentApp.openById(docId); // Get the body of the document var body = doc.getBody(); // Extract the text content var text = body.getText(); // Log the text content to the console Logger.log(text); // Example: Split the text into paragraphs var paragraphs = body.getParagraphs(); paragraphs.forEach(function(paragraph) { Logger.log(paragraph.getText()); }); }
